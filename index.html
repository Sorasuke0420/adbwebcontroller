<!-- WebUSB経由でADB通信を行うためのJavaScriptライブラリ（webadb等）を読み込む -->

<script src="https://cdn.jsdelivr.net/npm/webadb@0.1.1/webadb.js"></script>

<script>

let adb;

let shell;


// USB接続処理

document.getElementById('connectBtn').onclick = async () => {

try {

// Android ChromeがUSBデバイス（KYF36）を探索

const device = await Adb.open("webusb");

await device.connectAdb("host::");

adb = device;

alert("KYF36と接続成功！");

} catch (e) {

alert("接続エラー: " + e);

}

};


// コマンド送信共通関数

async function sendCommand(cmd) {

if (!adb) return alert("先にUSB接続してください");

let shell = await adb.shell(cmd);

await shell.receive(); // コマンド実行

}


// 文字送信

document.getElementById('sendTextBtn').onclick = () => {

let text = document.getElementById('textInput').value.replace(/ /g, '%s');

sendCommand(`input text "${text}"`);

document.getElementById('textInput').value = '';

};


// 十字キー（Keyevent）

document.getElementById('btnUp').onclick = () => sendCommand('input keyevent 19');

document.getElementById('btnDown').onclick = () => sendCommand('input keyevent 20');

document.getElementById('btnLeft').onclick = () => sendCommand('input keyevent 21');

document.getElementById('btnRight').onclick = () => sendCommand('input keyevent 22');

document.getElementById('btnCenter').onclick = () => sendCommand('input keyevent 23');

document.getElementById('btnBack').onclick = () => sendCommand('input keyevent 4');
